% \hold{...} underlines its argument (a syllable held on a longer note).  The
% rule sits a fixed \holddepth below the baseline -- just under the descenders,
% so underlines line up across syllables -- and is \holdthick thick (set
% independent of the mark size).  It is also \holdgap narrower than the syllable
% and centered, so the underlines of two adjacent held syllables stay separate
% rather than joining.  These dimensions are set from the font in layout.tex.
% The \rlap overlays the rule without consuming width; \leavevmode lets \hold
% start a line.
\newdimen\holddepth
\newdimen\holdthick
\newdimen\holdgap
\def\hold#1{%
  \leavevmode
  \setbox0=\hbox{#1}%
  \dimen0=\wd0 \advance\dimen0 by-\holdgap   % shorten the rule by a fixed amount
  \rlap{\kern.5\holdgap                       % ... and centre it under the syllable,
    \lower\holddepth\hbox{\vrule height0pt depth\holdthick width\dimen0}}%
  \box0                                        % leaving half the gap on each side
}
% \holdwide{...} is the mirror image of \hold: instead of falling \holdgap
% short of the syllable, its rule extends \holdgap beyond it on each side
% (still centered), for a syllable that should read as held even wider than
% its own width.
\def\holdwide#1{%
  \leavevmode
  \setbox0=\hbox{#1}%
  \dimen0=\wd0 \advance\dimen0 by\holdgap    % lengthen the rule by a fixed amount
  \rlap{\kern-.5\holdgap                      % ... and centre it under the syllable,
    \lower\holddepth\hbox{\vrule height0pt depth\holdthick width\dimen0}}%
  \box0                                        % extending half the gap on each side
}
% \holdexact{...} underlines the full width of the syllable, neither
% shortened (as \hold) nor lengthened (as \holdwide) -- no \holdgap and no
% centering kern needed, since the rule already matches the box exactly.
\def\holdexact#1{%
  \leavevmode
  \setbox0=\hbox{#1}%
  \rlap{\lower\holddepth\hbox{\vrule height0pt depth\holdthick width\wd0}}%
  \box0
}
